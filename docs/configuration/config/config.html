Compose an ekorpkit config¶

# %%capture
# %pip install ekorpkit

from ekorpkit import eKonf

cfg = eKonf.compose()
print('Config type:', type(cfg))

eKonf.pprint(cfg.dir)

cfg.env.dotenv_path = '.env'
eKonf.pprint(cfg.env)

cfg = eKonf.to_dict(cfg)
print(type(cfg))
