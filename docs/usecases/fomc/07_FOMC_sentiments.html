model_cfg = eKonf.compose('model/sentiment=lm')

cfg._pipeline_ = ['predict', 'aggregate_scores', 'replace', 'pivot', 'save_dataframe']

cfg.predict.path.output.filename = "fomc_sent_sentiments.parquet"
